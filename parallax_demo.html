// STATE: Subscribe to FZC-published fan speed (ECU → App)
val fanSpeed: Flow<FanSpeed> = climateCabinClient
    .fanSpeed()                          // STATE.VEHICLE.*.climate.cabin.fan_speed
    .map { it.value }

// COMMAND: Publish set request to EGW2 → FZC (App → ECU)
suspend fun setFanSpeed(level: Int) {
    climateCabinClient.setFanSpeed(           // SERVICE.VSVC.climate.cabin.set_fan_speed
        setFanSpeedRequest { fanSpeedLevel = level }
    )
    // → Elpis GW2 maps to VWG_XMM_HVAC_Requests PDU → FZC (BackboneETH, 100ms)
}

// STEP 1: App publishes set_drive_mode service request
client.setDriveMode(SetDriveModeRequest.newBuilder().setModeId(id).build())

// STEP 2: Orchestrator handles the request and maps to a recipe
override fun CoroutineScope.vehicleModelHandlers() {
    launch {
        vehicleDynamicsClient.handleSetDriveModeRequests().collect { req ->
            val recipe = DriveModeRecipeMapper.getRecipe(req.value.modeId)
            handleRequest(req, REQUEST_TIMEOUT) { applyRecipe(recipe) }
        }
    }
}

// STEP 3: applyRecipe() fan-outs to 6 sub-service calls → EGW2 → FZC/TVM
private suspend fun applyRecipe(recipe: DriveModeRecipe) {
    recipe.acceleratorPedalMapProfile?.let { vehicleDynamicsClient.setAccelPedalMapProfile(...) }
    recipe.regenBrakingGearDProfile?.let  { vehicleDynamicsClient.setRegenerativeBrakingProfile(...) }
    recipe.steeringFrontAxleProfile?.let  { vehicleDynamicsClient.setSteeringResponseProfile(...) }
    recipe.brakeTorqueVectoringProfile?.let { vehicleDynamicsClient.setTorqueDifferentialProfile(...) }
    recipe.tractionControlProfile?.let    { vehicleDynamicsClient.setTractionControlProfile(...) }
    recipe.escProfile?.let                { vehicleDynamicsClient.setElectronicStabilityControlProfile(...) }
    // Each call → Elpis GW2 → VDM_XMM_DrvMode* PDU → FZC/TVM (BackboneETH)
}
